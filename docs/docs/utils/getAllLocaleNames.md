# getAllLocaleNames

Returns a flat array of all available locale names from all regions.

```js
const BlizzAPI = require('blizzapi');

console.log(BlizzAPI.getAllLocaleNames());
// [
//   "en_US",
//   "es_MX",
//   "pt_BR",
//   "en_GB",
//   "es_ES",
//   "fr_FR",
//   "ru_RU",
//   "de_DE",
//   "pt_PT",
//   "it_IT",
//   "ko_KR",
//   "zh_TW",
//   "zh_CN"
// ]

```
